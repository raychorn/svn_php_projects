<?xml version="1.0" encoding="utf-8"?>
<mx:Canvas 
    xmlns:mx="http://www.adobe.com/2006/mxml" 
    width="100%" height="100%"
    >
    <mx:Script>
        <![CDATA[
            import mx.events.FlexEvent;
            import mx.controls.Image;
            import mx.controls.Spacer;
            import mx.containers.HBox;
            import mx.controls.HRule;
            import utils.StringUtils;
            import mx.controls.TextArea;
            import mx.controls.Label;
            import com.HAL.Alert.AlertPopUp;

            private var _dataProvider:Object = {};
            private var _labelField:String = 'label';
            private var _mbParent:DisplayObject;

            private var _rightAligned:Boolean = false;

            private var _debugger:TextArea;
            
            private static var _this:SmartMenuBarFlyout;

            [Embed(source="assets/images/icons/flyout-left.gif")]
            [Bindable]
            public var flyoutLeftIcon:Class;

            [Embed(source="assets/images/icons/flyout-right.gif")]
            [Bindable]
            public var flyoutRightIcon:Class;

            public function set dataProvider(dataProvider:*):void {
                if (dataProvider is Object) {
                    this._dataProvider = dataProvider;
                } else if (dataProvider) {
                    AlertPopUp.errorNoOkay('Cannot pass an object other than Object into the dataProvider setter for SmartMenuBarFlyout object.','ERROR');
                }
            }
            
            public function get dataProvider():Object {
                return this._dataProvider;
            }
            
            public function set labelField(labelField:String):void {
                if (this._labelField != labelField) {
                    this._labelField = labelField;
                }
            }
            
            public function get labelField():String {
                return this._labelField
            }
            
            public function set mbParent(mbParent:DisplayObject):void {
                this._mbParent = mbParent;
            }
            
            public function get mbParent():DisplayObject {
                return this._mbParent;
            }
            
            public function set debugger(debugger:TextArea):void {
                this._debugger = debugger;
            }
            
            public function get debugger():TextArea {
                return this._debugger;
            }
            
            public function set rightAligned(rightAligned:Boolean):void {
                this._rightAligned = rightAligned;
            }
            
            public function get rightAligned():Boolean {
                return this._rightAligned;
            }
            
            private function onCreationComplete_container():void {
                var aLabel:String = '';
                var anItem:Object;
                var anObj:*;
                var hbox:HBox;
                var vbox:VBox;
                var spacer:Spacer;
                var image:Image;
                SmartMenuBarFlyout._this = this;
                SmartMenuBarFlyout._this.height = 0;
                SmartMenuBarFlyout._this.width = 120;
                var isMenuItem:Boolean = false;
                try {
                    for (var i:String in this._dataProvider) {
                        anItem = this._dataProvider[i];
                        aLabel = anItem[this._labelField];
                        hbox = new HBox();
                        hbox.name = i;
                        hbox.percentWidth = 100;
                        vbox = new VBox();
                        vbox.percentHeight = 100;
                        spacer = new Spacer();
                        spacer.height = 1;
                        vbox.addChild(spacer);
                        image = new Image();
                        try { isMenuItem = (anItem.menuitem is Array) } catch (e:Error) { isMenuItem = false; }
                        image.source = (this._rightAligned) ? this.flyoutLeftIcon : this.flyoutRightIcon;
                        vbox.addChild(image);
                        if ( (isMenuItem) && (this._rightAligned) ) {
                            hbox.addChild(vbox);
                        }
                        if (aLabel is String) {
                            anObj = new Label();
                            anObj.htmlText = aLabel;
                            hbox.addChild(anObj);
                            hbox.styleName = 'myHBoxStyle';
                        } else {
                            spacer = new Spacer();
                            spacer.width = 1;
                            hbox.addChild(spacer);
                            anObj = new HRule();
                            anObj.width = this.container.width * 0.8;
                            hbox.addChild(anObj);
                        }
                        if ( (isMenuItem) && (!this._rightAligned) ) {
                            hbox.addChild(vbox);
                        }
                        hbox.addEventListener(MouseEvent.MOUSE_OVER,onMouseOver);
                        hbox.addEventListener(FlexEvent.CREATION_COMPLETE, function (event:FlexEvent):void { SmartMenuBarFlyout._this.height += event.currentTarget.getExplicitOrMeasuredHeight()+5; });
                        //hbox.addEventListener(FlexEvent.CREATION_COMPLETE, function (event:FlexEvent):void { SmartMenuBarFlyout._this.height += event.currentTarget.getExplicitOrMeasuredHeight()+5; SmartMenuBarFlyout._this.width = (event.currentTarget.getExplicitOrMeasuredWidth() > SmartMenuBarFlyout._this.width) ? event.currentTarget.getExplicitOrMeasuredWidth() : SmartMenuBarFlyout._this.width; });
                        this.container.addChild(hbox);
                        this._debugger.text = this.className + ' #' + i + ' :: ' + 'aLabel=' + aLabel + '\n' + this._debugger.text 
                    }
                    this._debugger.text = StringUtils.repeatedChars('-',40) + '\n' + this._debugger.text 
                } catch (e:Error) {}
            }
            
            private function onMouseOver(event:MouseEvent):void {
                try {
                    if (this._dataProvider is Array) {
                        var obj:HBox = event.currentTarget as HBox;
                        var i:int = Number(obj.name);
                        var a:Array = this._dataProvider as Array;
                        var data:Object = a[i];
                        var aLabel:String = '';
                        try { aLabel = data[this._labelField]; } catch (e:Error) {}
                        var menuitem:Array;
                        try { menuitem = data.menuitem; } catch (e:Error) {}
                        if (menuitem is Array) {
                            // open the next flyout panel...
                            var x:int = -1;
                        }
                        this._debugger.text = this.className + ' :: (onMouseOver) ' + ', rightAligned=' + this._rightAligned + ' (' + aLabel + ') ' + ' i=' + i + ' a.length=' + a.length + ' ' + ((menuitem is Array) ? '-->>' : '') + '\n' + this._debugger.text; 
                    }
                } catch (e:Error) { }
            }
        ]]>
    </mx:Script>
    <mx:VBox id="container" width="100%" height="100%" backgroundColor="white" borderColor="blue" borderStyle="solid" creationComplete="onCreationComplete_container()">
    </mx:VBox>
</mx:Canvas>
