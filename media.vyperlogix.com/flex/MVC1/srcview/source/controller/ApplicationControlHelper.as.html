package controller
{
    import model.ApplicationModel;
    
    /**
     * 
     * Manages all application control logic. Each event has an 
     * associated handling function which is invoked upon event 
     * dispatch via ApplicationController
     * 
     */        
    internal final class ApplicationControlHelper
    {  
        /**
         * 
         * Defines a reference to the ApplicationModel
         * 
         */
        private static var applicationModel:ApplicationModel = ApplicationModel.getInstance();
        
        /**
         * 
         * Handles an ApplicationEvent of event type EventTypes.SAY_HELLO_EVENT
         * 
         * @param ApplicationEvent object instance
         * 
         */        
        public static function handleSayHelloEvent(event:ApplicationEvent) : void
        {
            applicationModel.message = event.data.message;
        }
    }
}
