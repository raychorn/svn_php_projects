package controller
{
    import flash.events.Event;

    /**
     *  
     * Generic Event sub class which contains a data object associated with a 
     * specific event. The data object is used as the data payload of the 
     * event instance from which an event handler can utilize to gather required 
     * information specific to the event
     * 
     */
    public class ApplicationEvent extends Event 
    {
        /**
         * 
         * Defines a generic data payload for the event instance
         * 
         */
        public var data:Object;
        
        /**
         * 
         * Creates a new ApplicationEvent instance
         * 
         * @param Event type associated with the event instance
         * @param the data payload associated with the Event instance
         * 
         */        
        public function ApplicationEvent(type:String, data:Object)
        {
            super( type );
            this.data = data;
        }
    }
}
