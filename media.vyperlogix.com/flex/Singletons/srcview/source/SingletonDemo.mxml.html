            import Singletons.SimplerSingleton;
            
            [Bindable]
            private var someInfo:String = "Click the button.";
            
            private function clickMe(event:MouseEvent):void {
                SimplerSingleton.foo = 100;
                this.btn1.enabled = false;
                if (SimplerSingleton.foo.toString() == canvas1.txt_value.text) {
                    this.someInfo = "Very good.\n\nNotice the value shown above the 2nd button you clicked. This means there is only one instance of the SimplerSingleton because that one and only SimplerSingleton was successfully modified.\n\nThe SimplerSingleton was imported in the root and it was also imported into a Canvas MXML object.\n\nThere was no need to create any instances of the SimplerSingleton because there can be only one instance by definition.\n\nThe SimplerSingleton can contain any combination of variables and methods just as if the SimplerSingleton had been instantiated per the typical Singleton Pattern.";
                } else {
                    this.someInfo = "Oops.\n\nSomething went wrong !  Better restart the program or just refresh the web page...";
                }
            }
        ]]>
