                </HALGUI:LoginCanvas>
            </mx:AddChild>
            <mx:AddChild position="lastChild">
                <mx:Panel x="10" y="10" width="308" height="421" layout="absolute" id="panel_Info" title="INFO">
                    <mx:Text x="0" y="0" width="100%" height="100%" id="txt_info"/>
                </mx:Panel>
            </mx:AddChild>
        </mx:State>
        <mx:State name="LoggedInState">
            <mx:RemoveChild target="{canvas_NoAccess}"/>
            <mx:AddChild position="lastChild">
                <mx:Panel x="10" y="10" width="256" height="300" layout="absolute" id="panel_Info2" title="INFO :: LOGGED IN !">
                    <mx:VBox height="100%" width="100%">
                        <mx:VBox width="100%" height="80%" verticalAlign="top">

    <mx:Script>
        <![CDATA[
            import mx.collections.ArrayCollection;
            import mx.rpc.events.ResultEvent;
            import mx.rpc.events.FaultEvent;
            import com.HAL.Utils.EzObjectExplainer;
            import com.HAL.Alert.PopUp;
            import com.HAL.GUI.LoginRegisterWidgets.events.*;
            import com.HAL.util.EzHTTPService;
            
            [Bindable]
            public var isLoggedIn:Boolean = false;
            
            private var bool_EzHTTPService_DebugMode:Boolean = false;
    
            [Bindable]
            public var srvc:EzHTTPService = new EzHTTPService(bool_EzHTTPService_DebugMode);
            
            private const const_urlLoginRegister:String = "http://flash.1.ez-ajax.com/php/loginRegister/users.php";
            
            public function AlertShow(msg:String, title:String = ""):void {
                com.HAL.Alert.PopUp.error(((msg is String) ? msg : ""), title);
            }
            
            private function onFaultHTTPServiceConnector(event:FaultEvent):void {
                this.srvc.onFaultHandler(event);
            }

            private function onInit():void {
                this.srvc.alertShowFunc = this.AlertShow;
                this.srvc.faultCallBack = this.onFaultHTTPServiceConnector;
            }
            
            private function onCreationComplete():void {
                this.currentState = ((this.isLoggedIn) ? 'LoggedInState' : 'NotLoggedInState');
                this.txt_info.htmlText = "This User Login / User Registration sample uses the following components:<br><br><UL><LI>EzHTTPService</LI><LI>EzLoginWidget</LI></UL><br><br>The EzHTTPService component provides the connectivity with the database via an RPC REST Service and EzLoginWidget component provides the GUI.<br><br>LOGIN DEMO:<br><br>Username: Guest<br>Password: Guest";
            }

            private function onResultRegisterCompleted(event:ResultEvent):void {
                try {
                    var datum:ArrayCollection = this.srvc.makeEventResultIntoArrayCollection(event);
                    var id:int = datum.source[0].response.status.code;
                    var msg:String = datum.source[0].response.status.message;
                    id = ((id is int) ? id : 0);
                    msg = ((msg is String) ? msg : "");
                    if (id < 0) {
                        com.HAL.Alert.PopUp.error( msg.toString(), "Error Message");
                    } else {
                        this.currentState = 'RegisteredState';
                        this.txt_info3.htmlText = "Congrats !<br><br>You have been successfully Registered.";
                    }
                } catch (err:Error) {
                    com.HAL.Alert.PopUp.error( err.toString(), "Generic Error Message");
                }
            }

            private function onRegisterCompleted(event:RegisterCompletedEvent):void {
                try {
                    this.srvc.requestObj = new Object();
                    this.srvc.requestObj.cmd = "register";
                    this.srvc.requestObj.username = event.datum.username;
                    this.srvc.requestObj.password = event.datum.password;
                    this.srvc.requestObj.password2 = event.datum.password;    // repeated for the server-side logic - the GUI ensures the repeated password matches the password...
                    this.srvc.send(this.const_urlLoginRegister, onResultRegisterCompleted);
                } catch (err:Error) {
                    com.HAL.Alert.PopUp.error( err.toString(), "Generic Error Message :: onRegisterCompleted");
                }
            }
            
            private function onClickLogoutButton():void {
                this.currentState = 'NotLoggedInState';
                this.canvas_Login.currentState = '';
                this.canvas_Login.txt_Password.text = "";
            }
            
            private function onResultLoginRequested(event:ResultEvent):void {
                try {
                    var datum:ArrayCollection = this.srvc.makeEventResultIntoArrayCollection(event);
                    var id:int = datum.source[0].response.status.code;
                    var msg:String = datum.source[0].response.status.message;
                    id = ((id is int) ? id : 0);
                    msg = ((msg is String) ? msg : "");
                    if (id < 0) {
                        com.HAL.Alert.PopUp.error( msg.toString(), "Error Message");
                    } else {
                        this.currentState = 'LoggedInState';
                        this.txt_info2.htmlText = "Congrats !<br><br>You have successfully logged-in.";
                    }
                } catch (err:Error) {
                    com.HAL.Alert.PopUp.error( err.toString(), "Generic Error Message :: onResultLoginRequested");
                }
            }

            private function onLoginRequested(event:LoginRequestedEvent):void {
                try {
                    this.srvc.requestObj = new Object();
                    this.srvc.requestObj.cmd = "login";
                    this.srvc.requestObj.username = event.username;
                    this.srvc.requestObj.password = event.password;
                    this.srvc.send(this.const_urlLoginRegister, onResultLoginRequested);
                } catch (err:Error) {
                    com.HAL.Alert.PopUp.error( err.toString(), "Generic Error Message :: onLoginRequested");
                }
            }
            
            private function onCreationCompleteLoginCanvas():void {
                var url:String = Application.application.url;
                this.canvas_Login.url = url;
                this.canvas_Login.addEventListener(LoginRequestedEvent.TYPE_LOGIN_REQUESTED, onLoginRequested)
                this.canvas_Login.addEventListener(RegisterCompletedEvent.TYPE_REGISTER_COMPLETED, onRegisterCompleted)
            }
